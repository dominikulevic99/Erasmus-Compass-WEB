    <script>
        // Update score display when slider changes
        document.querySelectorAll('.slider').forEach(slider => {
            slider.addEventListener('input', function() {
                this.parentElement.querySelector('.value').textContent = this.value;
            });
        });

        // Update weight total in real-time
        document.querySelectorAll('.weight-input').forEach(input => {
            input.addEventListener('input', updateWeightTotal);
        });

        function updateWeightTotal() {
            const total = Array.from(document.querySelectorAll('.weight-input'))
                .reduce((sum, input) => sum + (parseInt(input.value) || 0), 0);
            document.getElementById('totalWeight').textContent = total;
        }

        function calculateResult() {
            const errorDiv = document.getElementById('errorMessage');
            errorDiv.style.display = 'none';

            // Collect scores
            const scores = {};
            document.querySelectorAll('.slider').forEach(slider => {
                scores[slider.name] = parseInt(slider.value);
            });

            // Collect weights
            const weights = {
                'Personal achievement development': parseInt(document.querySelector('input[name="pad-weight"]').value) || 0,
                'Financial': parseInt(document.querySelector('input[name="fin-weight"]').value) || 0,
                'Risk': parseInt(document.querySelector('input[name="risk-weight"]').value) || 0,
                'Psychological': parseInt(document.querySelector('input[name="psych-weight"]').value) || 0
            };

            // Validate total weight
            const totalWeight = Object.values(weights).reduce((a, b) => a + b, 0);
            if (totalWeight !== 100) {
                errorDiv.textContent = `❌ Weights must sum to 100 (current total: ${totalWeight})`;
                errorDiv.style.display = 'block';
                return;
            }

            // Send to backend
            fetch('/api/calculate', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ scores, weights })
            })
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    errorDiv.textContent = `❌ ${data.error}`;
                    errorDiv.style.display = 'block';
                    return;
                }
                displayResults(data);
            })
            .catch(error => {
                errorDiv.textContent = '❌ Error calculating results. Please try again.';
                errorDiv.style.display = 'block';
            });
        }

        function displayResults(data) {
            document.getElementById('utility').textContent = data.expected_utility;
            document.getElementById('percentage').textContent = data.utility_percentage;
            
            let recommendation = '';
            let recClass = '';
            if (data.utility_percentage >= 85) {
                recommendation = '✅ STRONG RECOMMENDATION: GO TO ERASMUS';
                recClass = 'strong';
            } else if (data.utility_percentage >= 70) {
                recommendation = '✅ POSITIVE RECOMMENDATION: ERASMUS IS PROBABLY WORTH IT';
                recClass = 'positive';
            } else if (data.utility_percentage >= 55) {
                recommendation = '⚠️ BALANCED DECISION: THINK CAREFULLY';
                recClass = 'balanced';
            } else if (data.utility_percentage >= 40) {
                recommendation = '⚠️ HIGH UNCERTAINTY: PREPARE MORE BEFORE DECIDING';
                recClass = 'uncertain';
            } else {
                recommendation = '❌ NOT RECOMMENDED RIGHT NOW';
                recClass = 'not-recommended';
            }
            
            const recDiv = document.getElementById('recommendation');
            recDiv.textContent = recommendation;
            recDiv.className = 'recommendation ' + recClass;
            document.getElementById('result').classList.add('show');
        }
    </script>
